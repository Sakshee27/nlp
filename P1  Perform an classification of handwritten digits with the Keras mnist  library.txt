from tensorflow import keras
from tensorflow.keras.layers import Input, Dense

img_tr[0].shape
input_layer=Input(shape=(784,))
hidden_layer1=Dense(522,activation="relu")
hidden_layer1=hidden_layer1(input_layer)
hidden_layer2=Dense(348,activation="relu")
hidden_layer2=hidden_layer2(hidden_layer1)
hidden_layer3=Dense(232,activation="relu")
hidden_layer3=hidden_layer3(hidden_layer2)
output_layer=Dense(10,activation="softmax")
output_layer=output_layer(hidden_layer3)
ann=keras.Model(inputs=input_layer,outputs=output_layer)
ann.summary()

ann.compile(optimizer="adam",loss="sparse_categorical_crossentropy",metrics=["accuracy"]
) 

es=keras.callbacks.EarlyStopping(monitor="loss",verbose=3,patience=1) 
img_tr_new=np.reshape(img_tr,(60000,784)) 
img_test_new=np.reshape(img_test,(10000,784)) 
ann.fit(img_tr_new,lbl_tr,validation_data=(img_test_new,lbl_test),epochs=200,callbacks=es)

img = img_test_new[0].reshape(1,784) 
p=ann.predict(img) 
p 

plist=[] 
for i in p: 
    plist.append(np.argmax(i)) 
print("Predicted Digit:", plist[0]) 
print("Actual Digit:", lbl_test[0])
