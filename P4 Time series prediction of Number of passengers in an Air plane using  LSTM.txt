def create_dataset_tensor(ds, seq_len):
    x_pytorch = []
    y_pytorch = []

    for i in range(len(ds) - seq_len):
        feature = ds[i:i+seq_len]
        target = ds[i+seq_len]
        x_pytorch.append(feature)
        y_pytorch.append(target)
    x_pytorch = np.array(x_pytorch)
    y_pytorch = np.array(y_pytorch)
    return torch.tensor(x_pytorch, dtype=torch.float32), torch.tensor(y_pytorch, dtype=torch.float32)

class AirModel(nn.Module):
    def __init__(self):
        super().__init__()
        self.input_layer = nn.Linear(in_features=1, out_features=32)
        self.lstm = nn.LSTM(input_size=32, hidden_size=64, num_layers=1, batch_first=True)
        # batch_first=True mean the order of dimension of x is (batch_size, sequence_len, features)
        self.output_layer = nn.Linear(64, 1)

    def forward(self, x):
        x = self.input_layer(x)
        output, (hidden_state, cell_state) = self.lstm(x)
        x = output[:, -1, :] #only passing the last timestep output, not sequence
        x = self.output_layer(x)
        return x
